309  uint8_t I2C_read(const uint8_t address, T& value) const {
320  uint8_t i{0}; // return number of bytes read
321  Wire.beginTransmission(MCP7940_ADDRESS); // Address the I2C device
322  Wire.write(address); // Send register address to read from
323  if (Wire.endTransmission() == 0) { // Close transmission and check error code
324  Wire.requestFrom(MCP7940_ADDRESS, sizeof(T)); // Request a block of data
325  uint8_t* bytePtr = (uint8_t*)&value; // Declare pointer to start of structure
326  for (i = 0; i < sizeof(T); i++) { // Loop for each byte to be read
327  *bytePtr++ = Wire.read(); // Read a byte
328  } // of for-next each byte
329  } // if-then success
330  return i; // return number of bytes read
331  } // end of template method "I2C_read"
332  template <typename T>
333  uint8_t I2C_write(const uint8_t address, const T& value) const {
343  Wire.beginTransmission(MCP7940_ADDRESS); // Address the I2C device
344  Wire.write(address); // Send register address to read from
345  Wire.write((uint8_t*)&value, sizeof(T)); // write the data
346  uint8_t i = Wire.endTransmission(); // close transmission and save status
347  if (i == 0) i = sizeof(T); // return number of bytes on success
348  return i; // return the number of bytes written
