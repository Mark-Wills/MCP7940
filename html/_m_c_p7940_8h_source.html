323  template <typename T>
324  uint8_t readEUI(const uint8_t& addr, T& value) const {
333  uint8_t i{0}; // return number of bytes read
334  Wire.beginTransmission(MCP7940_EUI_ADDRESS); // Address the special I2C address
335  Wire.write((addr % 8) + MCP7940_EUI_RAM_ADDRESS); // Send register address to read from
336  if (Wire.endTransmission() == 0) { // Close transmission and check error code
337  Wire.requestFrom(MCP7940_EUI_ADDRESS, sizeof(T)); // Request a block of data, max 61 bits
338  uint8_t* bytePtr = (uint8_t*)&value; // Declare pointer to start of structure
339  for (i = 0; i < sizeof(T); i++) { // Loop for each byte to be read
340  *bytePtr++ = Wire.read(); // Read a byte
341  } // of for-next each byte
342  } // if-then success
343  return i; // return number of bytes read
344  } // of method readEUI()
345  template <typename T>
346  uint8_t writeEUI(const uint8_t& addr, T& value) const {
356  uint8_t i{0}; // return number of bytes read
357  Wire.beginTransmission(MCP7940_EUI_ADDRESS); // Address the special I2C address
358  Wire.write(MCP7940_EEUNLOCK); // Send special register address to write to
359  Wire.write(0x55); // Special write value to start unlock
360  i = Wire.endTransmission(); // close transmission of first byte
361  Wire.beginTransmission(MCP7940_EUI_ADDRESS); // Address the special I2C address
362  Wire.write(MCP7940_EEUNLOCK); // Send special register address to write to
363  Wire.write(0x55); // Special write value to complete unlock
364  i = Wire.endTransmission(); // close transmission of second byte
365  Wire.beginTransmission(MCP7940_EUI_ADDRESS); // Address the special I2C address
366  Wire.write((addr % 8) + MCP7940_EUI_RAM_ADDRESS); // Send register address to read from
367  Wire.write((uint8_t*)&value, sizeof(T)); // write the data
368  i = Wire.endTransmission(); // close transmission of actual write
369  if (i == 0) i = sizeof(T); // return number of bytes on success
370  return i; // return number of bytes read
371  } // of method writeEUI()
