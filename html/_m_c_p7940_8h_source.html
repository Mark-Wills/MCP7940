253 /*******************************************************************************************************************
254 ** Declare the readRAM() and writeRAM() methods as template functions to use for all I2C device I/O. The code has **
255 ** to be in the main library definition rather than the actual MCP7940.cpp library file.The template functions **
256 ** allow any type of data to be read and written, be it a byte or a character array or a structure. **
257 ** **
258 ** The MCP7940 supports 64 bytes of general purpose SRAM memory, which can be used to store data. For more **
259 ** details, see datasheet page 36. **
260 ** **
261 ** The data is stored in a block of 64 bytes, reading beyond the end of the block causes the address pointer to **
262 ** roll over to the start of the block. **
263 *******************************************************************************************************************/
264 
265 /***************************************************************************************************************/
272  template< typename T >
273  uint8_t& readRAM(const uint8_t addr, T &value)
274  {
275  uint8_t* bytePtr = (uint8_t*)&value; // Pointer to structure beginning
276  uint8_t structSize = sizeof(T); // Number of bytes in structure
277  uint8_t i = 0; // loop counter
278  Wire.beginTransmission(MCP7940_ADDRESS); // Address the I2C device
279  Wire.write((addr%64) + MCP7940_RAM_ADDRESS); // Send register address to write
280  _TransmissionStatus = Wire.endTransmission(); // Close transmission
281  for (i=0;i<structSize;i++) // loop for each byte to be read
282  {
283  if (i%BUFFER_LENGTH==0) // loop for each buffer block
284  {
285  Wire.requestFrom(MCP7940_ADDRESS, structSize); // Request a block of data
286  } // of if-then we are at a buffer boundary
287  *bytePtr++ = Wire.read(); // next byte
288  } // of for-next each byte to be read
289  return (i);
290  } // of method readRAM()
291 /***************************************************************************************************************/
298  template<typename T>
299  bool writeRAM(const uint8_t addr, const T &value)
300  {
301  const uint8_t* bytePtr = (const uint8_t*)&value; // Pointer to structure beginning
302  Wire.beginTransmission(MCP7940_ADDRESS); // Address the I2C device
303  Wire.write((addr%64) + MCP7940_RAM_ADDRESS); // Send register address to write
304  for (uint8_t i = 0; i < sizeof(T); i++) // loop for each byte to be written
305  {
306  Wire.write(*bytePtr++);
307  } // of for-next each byte
308  _TransmissionStatus = Wire.endTransmission(); // Close transmission
309  return (!_TransmissionStatus); // return error status
310  } // of method writeRAM()
