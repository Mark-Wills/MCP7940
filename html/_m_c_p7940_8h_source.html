329  uint8_t readEUI(const uint8_t& addr, T& value) const {
338  uint8_t i{0}; // return number of bytes read
339  Wire.beginTransmission(MCP7940_EUI_ADDRESS); // Address the special I2C address
340  Wire.write((addr % 8) + MCP7940_EUI_RAM_ADDRESS); // Send register address to read from
341  if (Wire.endTransmission() == 0) { // Close transmission and check error code
342  Wire.requestFrom(MCP7940_EUI_ADDRESS, sizeof(T)); // Request a block of data, max 61 bits
343  uint8_t* bytePtr = (uint8_t*)&value; // Declare pointer to start of structure
344  for (i = 0; i < sizeof(T); i++) { // Loop for each byte to be read
345  *bytePtr++ = Wire.read(); // Read a byte
346  } // of for-next each byte
347  } // if-then success
348  return i; // return number of bytes read
349  } // of method readEUI()

351  uint8_t writeEUI(const uint8_t& addr, T& value) const {
361  uint8_t i{0}; // return number of bytes read
362  Wire.beginTransmission(MCP7940_EUI_ADDRESS); // Address the special I2C address
363  Wire.write(MCP7940_EEUNLOCK); // Send special register address to write to
364  Wire.write(0x55); // Special write value to start unlock

367  Wire.write(MCP7940_EEUNLOCK); // Send special register address to write to
368  Wire.write(0x55); // Special write value to complete unlock

372  Wire.write((uint8_t*)&value, sizeof(T)); // write the data

374  if (i == 0) i = sizeof(T); // return number of bytes on success
